Generate 4 independent Hypothesis tests for the sanitize_filename function from youtube_dl/utils.py. Produce complete pytest-compatible test code. **Generated tests** (paste here or reference the file in your repository):
```python
import os
import sys

from hypothesis import given, strategies as st

# Allow importing youtube_dl from local checkout
sys.path.insert(0, os.path.abspath("../../youtube-dl"))

from youtube_dl.utils import sanitize_filename


ILLEGAL_CHARS = '/\\:*?"<>|'


# 1. Property: sanitize_filename should never produce illegal filesystem characters
@given(st.text())
def test_no_illegal_characters_in_output(s):
    """
    Property: The output must not contain characters that are illegal in filenames.
    """
    result = sanitize_filename(s)

    assert isinstance(result, str)
    assert all(c not in result for c in ILLEGAL_CHARS)


# 2. Property: sanitization should be idempotent (applying it twice has no further effect)
@given(st.text())
def test_idempotence_of_sanitize_filename(s):
    """
    Property: sanitize_filename should be idempotent.
    """
    once = sanitize_filename(s)
    twice = sanitize_filename(once)

    assert once == twice


# 3. Property: Unicode and mixed-language input should be preserved except unsafe chars
@given(st.text(alphabet=st.characters()))
def test_unicode_preservation_except_illegal_chars(s):
    """
    Property: Unicode characters should be preserved unless they are illegal filename characters.
    """
    result = sanitize_filename(s)

    # Remove expected illegal characters from original and compare loosely
    expected = "".join(c for c in s if c not in ILLEGAL_CHARS)

    # sanitize_filename may also trim/normalize, so we only enforce containment rules
    for c in expected:
        assert c in result


# 4. Property: output should never be longer than input (since we only remove/replace characters)
@given(st.text())
def test_output_not_longer_than_input(s):
    """
    Property: sanitize_filename should not increase string length.
    """
    result = sanitize_filename(s)

    assert isinstance(result, str)
    assert len(result) <= len(s)

```
